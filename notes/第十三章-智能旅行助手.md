## 13.2 数据模型设计

### 13.2.1 为什么需要 Pydantic

**问题：** Python字典类型不安全
```python
# 字典方式 - 容易出错
trip = {
    "destination": "北京",
    "days": "3",  # 错误：应该是int
    "budget": None  # 缺少必需字段
}

# 访问时可能报错
total_cost = trip["budget"] * trip["days"]  # TypeError!
```

**Pydantic方式 - 类型安全**
```python
from pydantic import BaseModel, Field

class TripRequest(BaseModel):
    destination: str = Field(..., description="目的地城市")
    days: int = Field(..., gt=0, le=30, description="旅行天数")
    budget: float = Field(..., gt=0, description="预算（元）")
    interests: list[str] = Field(default=[], description="兴趣标签")

# 自动验证
try:
    trip = TripRequest(
        destination="北京",
        days="3",  # 自动转换为int
        budget=-1000  # 验证失败：必须>0
    )
except ValueError as e:
    print(e)  # "budget must be greater than 0"
```

## 13.3 多智能体协作设计

### 13.3.1 Agent 角色设计

**1. CoordinatorAgent（协调者）**
```python
class CoordinatorAgent:
    """总协调者，负责任务分配和结果整合"""
    
    def __init__(self, llm):
        self.llm = llm
        self.attraction_agent = AttractionAgent(llm)
        self.route_agent = RouteAgent(llm)
        self.budget_agent = BudgetAgent(llm)
    
    async def plan_trip(self, request: TripRequest) -> TravelPlan:
        """规划旅行"""
        
        # 1. 景点推荐
        attractions = await self.attraction_agent.recommend(
            destination=request.destination,
            interests=request.interests,
            days=request.days
        )
        
        # 2. 路线规划
        schedules = await self.route_agent.plan_route(
            attractions=attractions,
            days=request.days
        )
        
        # 3. 预算分配
        budget_plan = await self.budget_agent.allocate(
            schedules=schedules,
            total_budget=request.budget
        )
        
        # 4. 整合结果
        return TravelPlan(
            destination=request.destination,
            schedules=schedules,
            budget_breakdown=budget_plan,
            ...
        )
```

**2. AttractionAgent（景点专家）**
```python
class AttractionAgent:
    """景点推荐专家"""
    
    def __init__(self, llm, mcp_client):
        self.llm = llm
        self.mcp = mcp_client  # 高德地图MCP客户端
    
    async def recommend(self, destination, interests, days):
        """推荐景点"""
        
        # 1. 搜索景点
        pois = await self.mcp.call_tool(
            "search_pois",
            {
                "city": destination,
                "keywords": interests,
                "limit": days * 3  # 每天3个景点
            }
        )
        
        # 2. LLM筛选和排序
        prompt = f"""
        目的地：{destination}
        旅行天数：{days}
        用户兴趣：{interests}
        
        候选景点：
        {self._format_pois(pois)}
        
        请从以上景点中选择最合适的{days * 3}个，
        考虑因素：
        - 与用户兴趣的匹配度
        - 景点热度和评分
        - 地理位置的合理性
        
        输出JSON格式的景点列表。
        """
        
        selected = self.llm.generate(prompt)
        return self._parse_attractions(selected)
```

**3. RouteAgent（路线规划师）**
```python
class RouteAgent:
    """路线规划专家"""
    
    def __init__(self, llm, mcp_client):
        self.llm = llm
        self.mcp = mcp_client
    
    async def plan_route(self, attractions, days):
        """规划游览路线"""
        
        # 1. 计算景点间距离
        distance_matrix = await self._calculate_distances(attractions)
        
        # 2. 使用LLM进行智能分组
        prompt = f"""
        你是一个旅游路线规划专家。
        需要将{len(attractions)}个景点分配到{days}天的行程中。
        
        景点列表：
        {self._format_attractions(attractions)}
        
        距离矩阵：
        {distance_matrix}
        
        规划原则：
        1. 同一天的景点尽量靠近
        2. 每天游玩时间不超过8小时
        3. 考虑景点开放时间
        
        请给出每天的行程安排。
        """
        
        route_plan = self.llm.generate(prompt)
        
        # 3. 转换为结构化数据
        schedules = self._parse_schedules(route_plan, attractions)
        
        return schedules
    
    async def _calculate_distances(self, attractions):
        """计算景点间的距离和时间"""
        matrix = []
        for i, attr1 in enumerate(attractions):
            row = []
            for j, attr2 in enumerate(attractions):
                if i == j:
                    row.append(0)
                else:
                    # 调用高德API计算路径
                    result = await self.mcp.call_tool(
                        "calculate_route",
                        {
                            "origin": f"{attr1.longitude},{attr1.latitude}",
                            "destination": f"{attr2.longitude},{attr2.latitude}"
                        }
                    )
                    row.append(result["distance"])
            matrix.append(row)
        return matrix
```

## 13.4 MCP 工具集成详解

### 13.4.1 高德地图 MCP 集成

**创建MCP服务器：**
```python
# gaode_mcp_server.py
from mcp.server import Server
from mcp.types import Tool
import aiohttp

app = Server("gaode-map-server")

GAODE_API_KEY = "your_api_key"

@app.list_tools()
async def list_tools():
    return [
        Tool(
            name="search_pois",
            description="搜索POI（兴趣点）",
            inputSchema={
                "type": "object",
                "properties": {
                    "city": {"type": "string"},
                    "keywords": {"type": "string"},
                    "limit": {"type": "integer", "default": 10}
                },
                "required": ["city", "keywords"]
            }
        ),
        Tool(
            name="calculate_route",
            description="计算两点间路线",
            inputSchema={
                "type": "object",
                "properties": {
                    "origin": {"type": "string"},
                    "destination": {"type": "string"}
                },
                "required": ["origin", "destination"]
            }
        )
    ]

@app.call_tool()
async def call_tool(name: str, arguments: dict):
    if name == "search_pois":
        return await search_pois(**arguments)
    elif name == "calculate_route":
        return await calculate_route(**arguments)

async def search_pois(city, keywords, limit=10):
    """搜索POI"""
    url = "https://restapi.amap.com/v3/place/text"
    params = {
        "key": GAODE_API_KEY,
        "keywords": keywords,
        "city": city,
        "offset": limit
    }
    
    async with aiohttp.ClientSession() as session:
        async with session.get(url, params=params) as resp:
            data = await resp.json()
            return data["pois"]

async def calculate_route(origin, destination):
    """计算路线"""
    url = "https://restapi.amap.com/v3/direction/driving"
    params = {
        "key": GAODE_API_KEY,
        "origin": origin,
        "destination": destination
    }
    
    async with aiohttp.ClientSession() as session:
        async with session.get(url, params=params) as resp:
            data = await resp.json()
            route = data["route"]["paths"][0]
            return {
                "distance": route["distance"],
                "duration": route["duration"]
            }
```

**在Agent中使用：**
```python
from mcp import Client
from mcp.client.stdio import stdio_client

class AttractionAgent:
    def __init__(self):
        self.mcp_client = None
    
    async def __aenter__(self):
        # 启动MCP客户端
        server_params = StdioServerParameters(
            command="python",
            args=["gaode_mcp_server.py"]
        )
        self.read, self.write = await stdio_client(server_params).__aenter__()
        self.mcp_client = await Client(self.read, self.write).__aenter__()
        await self.mcp_client.initialize()
        return self
    
    async def search_attractions(self, city, keywords):
        result = await self.mcp_client.call_tool(
            name="search_pois",
            arguments={
                "city": city,
                "keywords": keywords,
                "limit": 20
            }
        )
        return result
```

          <el-checkbox label="艺术" />
        </el-checkbox-group>
      </el-form-item>
      
      <!-- 提交按钮 -->
      <el-form-item>
        <el-button type="primary" @click="submitForm" :loading="loading">
          生成行程
        </el-button>
      </el-form-item>
    </el-form>
  </div>

<script setup>
import { ref, reactive } from 'vue'
import { useRouter } from 'vue-router'
import axios from 'axios'

const router = useRouter()
const formRef = ref(null)
const loading = ref(false)

const form = reactive({
  destination: '',
  days: 3,
  budget: 5000,
  interests: []
})

const rules = {
  destination: [
    { required: true, message: '请输入目的地', trigger: 'blur' }
  ],
  days: [
    { required: true, message: '请选择天数', trigger: 'change' }
  ],
  budget: [
    { required: true, message: '请输入预算', trigger: 'blur' }
  ]
}

const submitForm = async () => {
  await formRef.value.validate()
  
  loading.value = true
  
  try {
    const response = await axios.post('/api/plan', form)
    
    // 跳转到结果页
    router.push({
      name: 'Result',
      params: { planId: response.data.id }
    })
  } catch (error) {
    console.error('规划失败:', error)
  } finally {
    loading.value = false
  }
}
</script>
