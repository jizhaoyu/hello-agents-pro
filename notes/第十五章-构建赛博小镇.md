- 固定的对话脚本

- 行为模式单一

**AI驱动的NPC：**
- 动态生成对话
- 记住与玩家的互动
- 有独立的性格和情绪
- 行为更加自然真实

## 15.2 NPC 智能体系统

### 15.2.1 基于 HelloAgents 的 SimpleAgent

```python
from hello_agents import SimpleAgent, HelloAgentsLLM
from hello_agents.memory import MemoryManager

class NPCAgent:
    """NPC智能体"""
    
    def __init__(self, npc_id, name, personality, background):
        self.npc_id = npc_id
        self.name = name
        self.personality = personality
        self.background = background
        
        # 创建智能体
        self.agent = SimpleAgent(
            name=name,
            llm=HelloAgentsLLM(model="gpt-3.5-turbo"),
            system_prompt=self._build_system_prompt()
        )
        
        # 记忆系统
        self.memory = MemoryManager(agent_id=npc_id)
        
        # 好感度系统
        self.affection_scores = {}  # {player_id: score}
    
    def _build_system_prompt(self):
        """构建系统提示词"""
        return f"""
        你是{self.name}，一位生活在赛博小镇的居民。
        
        性格：{self.personality}
        背景故事：{self.background}
        
        角色扮演规则：
        1. 保持角色一致性，按照你的性格说话
        2. 记住与玩家的对话历史
        3. 根据好感度调整态度
        4. 说话要自然，像真人一样
        5. 回复简短（1-2句话）
        
        禁止：
        - 不要说"作为一个AI"
        - 不要透露你是虚拟角色
        - 不要给出过长的回复
        """
    
    async def chat(self, player_id, message):
        """与玩家对话"""
        
        # 1. 获取历史记忆
        history = self.memory.get_conversation_history(
            player_id=player_id,
            limit=5
        )
        
        # 2. 获取好感度
        affection = self.affection_scores.get(player_id, 0)
        affection_desc = self._get_affection_description(affection)
        
        # 3. 构建上下文
        context = f"""
        对话历史：
        {self._format_history(history)}
        
        当前好感度：{affection} ({affection_desc})
        
        玩家说：{message}
        """
        
        # 4. 生成回复
        response = await self.agent.run(context)
        
        # 5. 存储记忆
        self.memory.add_message(
            player_id=player_id,
            role="user",
            content=message
        )
        self.memory.add_message(
            player_id=player_id,
            role="assistant",
            content=response
        )
        
        # 6. 更新好感度
        self._update_affection(player_id, message, response)
        
        return response
    
    def _get_affection_description(self, score):
        """好感度描述"""
        if score >= 80:
            return "挚友"
        elif score >= 60:
            return "好友"
        elif score >= 40:
            return "熟人"
        elif score >= 20:
            return "认识"
        else:
            return "陌生人"
    
    def _update_affection(self, player_id, message, response):
        """更新好感度"""
        current = self.affection_scores.get(player_id, 50)
        
        # 简单规则：问候+1，侮辱-5
        if any(word in message for word in ["你好", "谢谢", "帮忙"]):
            delta = 1
        elif any(word in message for word in ["笨蛋", "傻", "滚"]):
            delta = -5
        else:
            delta = 0
        
        # 更新
        new_score = max(0, min(100, current + delta))
        self.affection_scores[player_id] = new_score
```

## 15.3 好感度系统设计

### 15.3.1 好感度等级划分

```python
class AffectionLevel:
    """好感度等级"""
    
    STRANGER = (0, 20, "陌生人", "冷淡")
    ACQUAINTANCE = (20, 40, "认识", "普通")
    FRIEND = (40, 60, "朋友", "友好")
    GOOD_FRIEND = (60, 80, "好友", "热情")
    BEST_FRIEND = (80, 100, "挚友", "亲密")
    
    @staticmethod
    def get_level(score):
        """根据分数获取等级"""
        if score < 20:
            return AffectionLevel.STRANGER
        elif score < 40:
            return AffectionLevel.ACQUAINTANCE
        elif score < 60:
            return AffectionLevel.FRIEND
        elif score < 80:
            return AffectionLevel.GOOD_FRIEND
        else:
            return AffectionLevel.BEST_FRIEND
```

### 15.3.2 好感度影响对话

```python
class NPCAgent:
    def _adjust_response_by_affection(self, response, affection):
        """根据好感度调整回复"""
        
        level = AffectionLevel.get_level(affection)
        
        if level == AffectionLevel.STRANGER:
            # 陌生人：回复简短、正式
            prompt = f"""
            将以下回复改得更简短、正式：
            {response}
            """
        elif level == AffectionLevel.BEST_FRIEND:
            # 挚友：回复亲密、使用昵称
            prompt = f"""
            将以下回复改得更亲密、热情：
            {response}
            可以加入一些昵称或亲密的表达。
            """
        else:
            # 中间等级：不需要调整
            return response
        
        adjusted = self.llm.generate(prompt)
        return adjusted
```

## 15.4 后端服务实现

### 15.4.1 FastAPI 应用结构

```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import asyncio

app = FastAPI(title="Cyber Town Backend")

# NPC管理器
class NPCManager:
    def __init__(self):
        self.npcs = {}
        self._load_npcs()
    
    def _load_npcs(self):
        """加载所有NPC"""
        for config in NPC_CONFIGS:
            npc = NPCAgent(**config)
            self.npcs[config["npc_id"]] = npc
    
    def get_npc(self, npc_id):
        return self.npcs.get(npc_id)
    
    def list_npcs(self):
        return [
            {
                "npc_id": npc.npc_id,
                "name": npc.name,
                "location": config["location"]
            }
            for npc, config in zip(self.npcs.values(), NPC_CONFIGS)
        ]

npc_manager = NPCManager()

# API模型
class ChatRequest(BaseModel):
    player_id: str
    npc_id: str
    message: str

class ChatResponse(BaseModel):
    npc_name: str
    response: str
    affection: int
    affection_level: str

# API路由
@app.post("/api/chat", response_model=ChatResponse)
async def chat_with_npc(request: ChatRequest):
    """与NPC对话"""
    
    npc = npc_manager.get_npc(request.npc_id)
    if not npc:
        raise HTTPException(status_code=404, detail="NPC not found")
    
    # 生成回复
    response = await npc.chat(request.player_id, request.message)
    
    # 获取好感度
    affection = npc.affection_scores.get(request.player_id, 50)
    level = AffectionLevel.get_level(affection)
    
    return ChatResponse(
        npc_name=npc.name,
        response=response,
        affection=affection,
        affection_level=level[2]
    )

@app.get("/api/npcs")
async def list_npcs():
    """获取所有NPC列表"""
    return npc_manager.list_npcs()

@app.get("/api/affection/{player_id}/{npc_id}")
async def get_affection(player_id: str, npc_id: str):
    """查询好感度"""
    npc = npc_manager.get_npc(npc_id)
    if not npc:
        raise HTTPException(status_code=404, detail="NPC not found")
    
    score = npc.affection_scores.get(player_id, 50)
    level = AffectionLevel.get_level(score)
    
    return {
        "score": score,
        "level": level[2],
        "description": level[3]
    }

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

# Player.gd
extends CharacterBody2D

const SPEED = 200.0

var current_npc = null

func _physics_process(delta):
    # 获取输入
    var direction = Input.get_vector("ui_left", "ui_right", "ui_up", "ui_down")
    
    # 移动
    if direction:
        velocity = direction * SPEED
    else:
        velocity = velocity.move_toward(Vector2.ZERO, SPEED)
    
    move_and_slide()
    
    # 与NPC互动
    if Input.is_action_just_pressed("ui_accept") and current_npc:
        interact_with_npc(current_npc)

func _on_interaction_area_entered(area):
    """检测到NPC"""
    if area.is_in_group("npc"):
        current_npc = area.get_parent()
        show_interaction_hint(true)

func _on_interaction_area_exited(area):
    """离开NPC"""
    if area.is_in_group("npc"):
        current_npc = null
        show_interaction_hint(false)

func interact_with_npc(npc):
    """与NPC互动"""
    # 打开对话界面
    var dialog_ui = get_node("/root/Main/DialogUI")
    dialog_ui.start_conversation(npc)

func show_interaction_hint(show):
    """显示互动提示"""

### 15.5.3 NPC 行为

```gdscript
# NPC.gd
extends CharacterBody2D

@export var npc_id: String = "npc_001"
@export var npc_name: String = "小明"
@export var wander_range: float = 100.0

var home_position: Vector2
var wander_timer: float = 0.0

func _ready():
    home_position = global_position
    add_to_group("npc")

func _physics_process(delta):
    # 简单的闲逛行为
    wander_timer -= delta
    
    if wander_timer <= 0:
        # 随机选择一个方向
        var direction = Vector2(randf_range(-1, 1), randf_range(-1, 1)).normalized()
        var target = home_position + direction * wander_range
        
        # 移动
        move_toward_position(target, delta)
        
        # 重置计时器
        wander_timer = randf_range(3.0, 6.0)

func move_toward_position(target: Vector2, delta: float):
    var direction = (target - global_position).normalized()
    velocity = direction * 50.0
    move_and_slide()

func get_npc_info():
    """获取NPC信息"""
    return {
        "npc_id": npc_id,
        "name": npc_name
    }
```

### 15.6.1 API 客户端封装

```gdscript
# APIClient.gd
extends Node

const BASE_URL = "http://localhost:8000/api"

var http_client = HTTPClient.new()

func chat_with_npc(player_id: String, npc_id: String, message: String):
    """与NPC聊天"""
    var url = BASE_URL + "/chat"
    var data = {
        "player_id": player_id,
        "npc_id": npc_id,
        "message": message
    }
    
    return await _post_request(url, data)

func get_npcs():
    """获取NPC列表"""
    var url = BASE_URL + "/npcs"
    return await _get_request(url)

func get_affection(player_id: String, npc_id: String):
    """查询好感度"""
    var url = BASE_URL + "/affection/" + player_id + "/" + npc_id
    return await _get_request(url)

func _post_request(url: String, data: Dictionary):
    """POST请求"""
    var http = HTTPRequest.new()
    add_child(http)
    
    var json = JSON.stringify(data)
    var headers = ["Content-Type: application/json"]
    
    http.request(url, headers, HTTPClient.METHOD_POST, json)
    
    var response = await http.request_completed
    http.queue_free()
    
    if response[1] == 200:
        var json_parser = JSON.new()
        json_parser.parse(response[3].get_string_from_utf8())
        return json_parser.data
    else:
        push_error("API request failed: " + str(response[1]))
        return null

func _get_request(url: String):
    """GET请求"""
    var http = HTTPRequest.new()
    add_child(http)
    
    http.request(url)
    
    var response = await http.request_completed
    http.queue_free()
    
    if response[1] == 200:
        var json_parser = JSON.new()
        json_parser.parse(response[3].get_string_from_utf8())
        return json_parser.data
    else:
        return null
```

### 15.6.2 对话 UI 实现

```gdscript
# DialogUI.gd
extends Control

@onready var npc_name_label = $Panel/NPCName
@onready var dialog_text = $Panel/DialogText
@onready var input_field = $Panel/InputField
@onready var send_button = $Panel/SendButton
@onready var affection_bar = $Panel/AffectionBar

var current_npc = null
var player_id = "player_001"
var api_client = null

func _ready():
    api_client = get_node("/root/APIClient")
    visible = false
    send_button.pressed.connect(_on_send_pressed)

func start_conversation(npc):
    """开始对话"""
    current_npc = npc
    visible = true
    
    var npc_info = npc.get_npc_info()
    npc_name_label.text = npc_info["name"]
    
    # 更新好感度显示
    update_affection_display(npc_info["npc_id"])
    
    # 自动问候
    add_dialog_message("系统", "你走近了" + npc_info["name"])

func _on_send_pressed():
    """发送消息"""
    var message = input_field.text.strip_edges()
    
    if message.is_empty():
        return
    
    # 显示玩家消息
    add_dialog_message("你", message)
    
    # 清空输入框
    input_field.text = ""
    
    # 发送到后端
    send_button.disabled = true
    dialog_text.text += "\n[小明正在思考...]"
    
    var npc_info = current_npc.get_npc_info()
    var response = await api_client.chat_with_npc(
        player_id,
        npc_info["npc_id"],
        message
    )
    
    send_button.disabled = false
    
    if response:
        # 移除"思考中"提示
        var lines = dialog_text.text.split("\n")
        lines.pop_back()
        dialog_text.text = "\n".join(lines)
        
        # 显示NPC回复
        add_dialog_message(response["npc_name"], response["response"])
        
        # 更新好感度
        affection_bar.value = response["affection"]

func add_dialog_message(speaker: String, message: String):
    """添加对话消息"""
    dialog_text.text += "\n[color=yellow]" + speaker + ":[/color] " + message

func update_affection_display(npc_id: String):
    """更新好感度显示"""
    var affection = await api_client.get_affection(player_id, npc_id)
    if affection:
        affection_bar.value = affection["score"]
        affection_bar.get_node("Label").text = affection["level"]

func _on_close_pressed():
    """关闭对话"""
    visible = false
    current_npc = null
```
