## 📚 项目概述

### 14.1.1 为什么需要深度研究助手

**传统研究流程的痛点：**
1. 信息收集耗时（Google搜索→阅读→整理）
2. 需要多轮迭代（发现新问题→再次搜索）
3. 笔记分散混乱（难以回顾和总结）
4. 缺乏系统性（容易遗漏关键信息）

**智能研究助手的价值：**
- 自动规划研究任务
- 多源信息检索和整合
- 结构化笔记记录
- 生成完整的研究报告

## 14.2 TODO 驱动的研究范式

### 14.2.1 什么是 TODO 驱动的研究

**核心思想：** 将研究任务分解为一系列待办事项（TODO），智能体逐个完成并动态调整。

**类比：** 就像你做项目管理
1. 列出任务清单
2. 逐个完成任务
3. 完成过程中发现新任务→加入清单
4. 直到所有任务完成

**与传统方法的对比：**

```python
# 传统方法：固定流程
def traditional_research(topic):
    # 1. 搜索一次
    results = search(topic)
    
    # 2. 总结
    summary = summarize(results)
    
    # 3. 结束
    return summary

# 问题：信息可能不够深入，无法根据发现调整


# TODO驱动：动态迭代
def todo_driven_research(topic):
    # 1. 生成初始TODO
    todo_list = plan_research(topic)
    
    # 2. 循环执行
    while todo_list:
        # 取出一个TODO
        current_todo = todo_list.pop(0)
        
        # 执行研究
        result = research(current_todo)
        
        # 记录笔记
        notes.append(result)
        
        # 发现新问题？添加新TODO
        new_todos = discover_questions(result)
        todo_list.extend(new_todos)
    
    # 3. 整合所有笔记
    return generate_report(notes)
```

### 14.2.2 三阶段研究流程

**阶段1：规划（Planning）**
```python
class PlannerAgent:
    """规划研究任务"""
    
    def plan(self, topic):
        prompt = f"""
        你是一位研究规划专家。
        用户想要深入了解：{topic}
        
        请制定一个系统的研究计划，分解为具体的TODO：
        
        要求：
        1. 从基础概念开始，逐步深入
        2. 每个TODO是一个明确的问题
        3. 考虑多个维度（定义、原理、应用、挑战等）
        4. 一般5-10个TODO即可
        
        输出JSON格式：
        [
          {{"id": 1, "question": "什么是X？", "priority": "high"}},
          {{"id": 2, "question": "X的原理是什么？", "priority": "high"}},
          ...
        ]
        """
        
        todos = self.llm.generate(prompt)
        return json.loads(todos)
```

**阶段2：研究（Researching）**
```python
class ResearchAgent:
    """执行研究任务"""
    
    def __init__(self):
        self.search_tool = MultiSourceSearchTool()
        self.note_tool = NoteTool()
    
    async def research_todo(self, todo):
        """研究单个TODO"""
        
        # 1. 多源搜索
        search_results = await self.search_tool.search(
            query=todo["question"],
            sources=["google", "wikipedia", "arxiv"]
        )
        
        # 2. 分析和总结
        analysis = self.analyze(todo["question"], search_results)
        
        # 3. 记录笔记
        self.note_tool.create_section(todo["question"])
        self.note_tool.append(analysis)
        
        # 4. 发现新问题
        new_questions = self.discover_followup(analysis)
        
        return {
            "todo_id": todo["id"],
            "answer": analysis,
            "new_questions": new_questions,
            "sources": [r["url"] for r in search_results]
        }
    
    def analyze(self, question, search_results):
        """分析搜索结果"""
        
        # 整合多个来源的信息
        context = "\n\n".join([
            f"来源 {i+1}: {r['content'][:500]}..."
            for i, r in enumerate(search_results[:3])
        ])
        
        prompt = f"""
        问题：{question}
        
        参考资料：
        {context}
        
        请基于以上资料，深入回答这个问题。
        
        要求：
        1. 准确、客观
        2. 引用具体来源
        3. 如果资料不足，明确指出
        4. 提出可能需要进一步研究的问题
        """
        
        return self.llm.generate(prompt)
    
    def discover_followup(self, analysis):
        """从分析中发现后续问题"""
        
        prompt = f"""
        基于以下研究内容：
        {analysis}
        
        请列出3个值得进一步深入研究的问题。
        
        输出JSON数组格式。
        """
        
        questions = self.llm.generate(prompt)
        return json.loads(questions)
```

**阶段3：总结（Summarizing）**
```python
class SummarizerAgent:
    """生成研究报告"""
    
    def generate_report(self, note_tool):
        """基于笔记生成最终报告"""
        
        # 1. 获取所有笔记
        full_notes = note_tool.export()
        
        # 2. 生成结构化报告
        prompt = f"""
        基于以下研究笔记，生成一份完整的研究报告。
        
        研究笔记：
        {full_notes}
        
        报告结构：
        # 标题
        
        ## 摘要
        (200字概述)
        
        ## 核心发现
        (列举3-5个关键点)
        
        ## 详细分析
        (按主题组织内容)
        
        ## 结论
        
        ## 参考资料
        (列出所有引用来源)
        """
        
        report = self.llm.generate(prompt)
        return report
```

## 14.3 智能体系统设计

### 14.3.1 完整系统实现

```python
class DeepResearchAssistant:
    """深度研究助手主控制器"""
    
    def __init__(self, llm):
        self.planner = PlannerAgent(llm)
        self.researcher = ResearchAgent(llm)
        self.summarizer = SummarizerAgent(llm)
        self.note_tool = NoteTool()
    
    async def research(self, topic, max_iterations=20):
        """执行完整的研究流程"""
        
        print(f"🔬 开始研究: {topic}")
        
        # ===== 阶段1: 规划 =====
        print("\n📋 生成研究计划...")
        todo_list = self.planner.plan(topic)
        print(f"生成了 {len(todo_list)} 个研究任务")
        
        # ===== 阶段2: 研究 =====
        print("\n🔍 开始深度研究...")
        
        iteration = 0
        completed_todos = []
        
        while todo_list and iteration < max_iterations:
            iteration += 1
            
            # 取出优先级最高的TODO
            todo_list.sort(key=lambda x: self._priority_score(x))
            current_todo = todo_list.pop(0)
            
            print(f"\n[{iteration}] 研究问题: {current_todo['question']}")
            
            # 执行研究
            result = await self.researcher.research_todo(current_todo)
            completed_todos.append(result)
            
            print(f"   ✓ 已完成，找到 {len(result['sources'])} 个来源")
            
            # 添加新发现的问题
            if result['new_questions']:
                new_todos = [
                    {
                        "id": len(todo_list) + len(completed_todos) + i,
                        "question": q,
                        "priority": "low"
                    }
                    for i, q in enumerate(result['new_questions'])
                ]
                
                # 去重
                existing_questions = {t['question'] for t in todo_list}
                new_todos = [t for t in new_todos if t['question'] not in existing_questions]
                
                if new_todos:
                    todo_list.extend(new_todos)
                    print(f"   + 新增 {len(new_todos)} 个待研究问题")
        
        # ===== 阶段3: 总结 =====
        print("\n📝 生成研究报告...")
        report = self.summarizer.generate_report(self.note_tool)
        
        print("\n✅ 研究完成！")
        print(f"   - 完成任务数: {len(completed_todos)}")
        print(f"   - 迭代次数: {iteration}")
        print(f"   - 笔记字数: {len(self.note_tool.export())}")
        
        return {
            "report": report,
            "completed_todos": completed_todos,
            "notes": self.note_tool.export()
        }
    
    def _priority_score(self, todo):
        """计算优先级分数"""
        priority_map = {"high": 3, "medium": 2, "low": 1}
        return -priority_map.get(todo.get("priority", "medium"), 2)
```

## 14.4 工具系统集成

### 14.4.1 多源搜索工具

```python
class MultiSourceSearchTool:
    """多源搜索工具"""
    
    def __init__(self):
        self.sources = {
            "google": GoogleSearchAPI(),
            "wikipedia": WikipediaAPI(),
            "arxiv": ArxivAPI(),
            "github": GitHubSearchAPI()
        }
    
    async def search(self, query, sources=None, top_k=5):
        """从多个来源搜索"""
        
        if sources is None:
            sources = ["google", "wikipedia"]
        
        all_results = []
        
        # 并行搜索多个来源
        tasks = [
            self.sources[source].search(query, limit=top_k)
            for source in sources
            if source in self.sources
        ]
        
        results_list = await asyncio.gather(*tasks)
        
        # 整合结果
        for source, results in zip(sources, results_list):
            for result in results:
                result["source"] = source
                all_results.append(result)
        
        # 去重和排序
        all_results = self._deduplicate(all_results)
        all_results = self._rank_by_relevance(query, all_results)
        
        return all_results[:top_k]
    
    def _deduplicate(self, results):
        """去重"""
        seen_urls = set()
        unique_results = []
        
        for result in results:
            url = result.get("url", "")
            if url not in seen_urls:
                seen_urls.add(url)
                unique_results.append(result)
        
        return unique_results
    
    def _rank_by_relevance(self, query, results):
        """按相关性排序"""
        # 简单实现：基于标题和内容的关键词匹配
        keywords = set(query.lower().split())
        
        for result in results:
            text = (result.get("title", "") + " " + result.get("content", "")).lower()
            score = sum(1 for kw in keywords if kw in text)
            result["relevance_score"] = score
        
        return sorted(results, key=lambda x: x["relevance_score"], reverse=True)
```

```vue
<script setup>
import { ref } from 'vue'
import axios from 'axios'
import MarkdownIt from 'markdown-it'

const md = new MarkdownIt()

const topic = ref('')
const researching = ref(false)
const currentTask = ref('')
const completedTasks = ref([])
const pendingCount = ref(0)
const report = ref('')

const startResearch = async () => {
  researching.value = true
  completedTasks.value = []
  
  // 建立SSE连接接收实时进度

    `/api/research/stream?topic=${encodeURIComponent(topic.value)}`

    researching.value = false

  a.download = 'research_report.md'
```
